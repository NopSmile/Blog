static bool ReadFile(const char *file, std::string &out)
{
    std::ifstream fin(file);
    if (!fin.is_open())
    {
        return false;
    }
​

    buffer << fin.rdbuf();
    out.assign(buffer.str());
    fin.close();
​
    return true;

            now = int(time.time())
            self._file_content = f.read()
            self._file_last_mtime = now
            self._file_last_load_time = now
